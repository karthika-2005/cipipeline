pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/karthika-2005/cipipeline.git'
            }
        }

        stage('Download Artifact from CI') {
            steps {
                copyArtifacts(
                    projectName: 'cipipeline',   // name of your CI job
                    selector: lastSuccessful()
                )
            }
        }

        stage('Deploy') {
            steps {
                bat 'java -jar target/ciex-app-1.5.jar'
            }
        }
    }
}
